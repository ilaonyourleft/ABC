*** PARALLEL ***

--- compile program
-O optimize
-o <file> where to put the executable
-l link library
(mpicc -o myprog myprog.c lib.o)
mpicc -o ABC_parallel ABC_parallel.c ABC_sequential_lib.o

--- run program
-np <p> number of processes
<exec> executable
<arg1> ... command-line arguments
(mpirun -np <p> <exec> <arg1> ...)
mpirun -np 1 ABC_parallel
